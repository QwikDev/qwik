export interface TransformModulesOptions {
  input: TransformModuleInput[];

  mode?: EmitMode;
  scope?: string;
  stripExports?: string[];
  regCtxName?: string[];
  stripCtxName?: string[];
  stripEventHandlers?: boolean;
  isServer?: boolean;
}

export interface TransformModuleInput {
  path: string;
  code: string;
  devPath?: string;
}

export interface TransformOutput {
  modules: TransformModule[];
  diagnostics: Diagnostic[];
  isTypeScript: boolean;
  isJsx: boolean;
}

export interface TransformModule {
  path: string;
  isEntry: boolean;
  code: string;
  map: string | null;
  segment: SegmentAnalysis | null;
  origPath: string | null;

export interface SegmentAnalysis {
  origin: string;
  name: string;
  entry: string | null;
  displayName: string;
  hash: string;
  canonicalFilename: string;
  extension: string;
  parent: string | null;
  ctxKind: 'eventHandler' | 'function';
  ctxName: string;
  captures: boolean;
  loc: [number, number];
}

export type EntryStrategy =
  | { type: 'inline' }
  | { type: 'hoist' }
  | { type: 'hook'; manual?: Record<string, string> }
  | { type: 'segment'; manual?: Record<string, string> }
  | { type: 'single'; manual?: Record<string, string> }
  | { type: 'component'; manual?: Record<string, string> }
  | { type: 'smart'; manual?: Record<string, string> };

export type MinifyMode = 'simplify' | 'none';
export type EmitMode = 'dev' | 'prod' | 'lib';

export interface Diagnostic {
  severity: 'error' | 'warning';
  code: string;
  message: string;
  file: string;
  highlights: DiagnosticHighlight[];
}

export interface DiagnosticHighlight {
  message: string | null;
  loc: SourceLocation;
}

export interface SourceLocation {
  start_line: number;
  start_col: number;
  end_line: number;
  end_col: number;
}

Create `src/optimizer/rewrite-imports.ts` implementing import source path rewriting per IMP-01, IMP-02, IMP-03.

The rewrite rules (verified from rename_builder_io snapshot):
```typescript
const IMPORT_REWRITES: [string, string][] = [
  ['@builder.io/qwik', '@qwik.dev/core'],
  ['@builder.io/qwik-city', '@qwik.dev/router'],
  ['@builder.io/qwik-react', '@qwik.dev/react'],
];
```

Export `rewriteImportSource(source: string): string` that:
1. Checks if `source` equals or starts with one of the old package names (followed by `/`)
2. If match: replaces the base package portion with the new name, preserving any sub-path
3. If no match: returns source unchanged
4. Order matters: check `@builder.io/qwik-city` before `@builder.io/qwik` to avoid false prefix match, OR use exact-then-subpath matching

Example mappings:
- `@builder.io/qwik` -> `@qwik.dev/core`
- `@builder.io/qwik/build` -> `@qwik.dev/core/build`
- `@builder.io/qwik/jsx-runtime` -> `@qwik.dev/core/jsx-runtime`
- `@builder.io/qwik-city` -> `@qwik.dev/router`
- `@builder.io/qwik-city/more/here` -> `@qwik.dev/router/more/here`
- `@builder.io/qwik-react` -> `@qwik.dev/react`
- `@builder.io/sdk` -> `@builder.io/sdk` (NO rewrite -- not a qwik package)
- `react` -> `react` (NO rewrite)

IMPORTANT: Sort IMPORT_REWRITES by descending length of the `from` key so longer prefixes match first (e.g., `@builder.io/qwik-city` before `@builder.io/qwik`). This prevents `@builder.io/qwik-city` from matching the `@builder.io/qwik` rule.
