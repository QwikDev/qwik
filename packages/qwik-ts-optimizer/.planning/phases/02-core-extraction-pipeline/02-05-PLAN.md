  <action>
Create `src/optimizer/transform.ts` as the public entry point for the optimizer.

**Main export:**
```typescript
export function transformModule(options: TransformModulesOptions): TransformOutput;
```

**Algorithm:**
1. For each input in `options.input`:
   a. Determine `relPath` from `input.path` relative to `options.srcDir`
   b. Determine `isTypeScript` from extension (.ts, .tsx)
   c. Determine `isJsx` from extension (.tsx, .jsx)
   d. Call `extractSegments(input.code, relPath, options.scope)`
   e. Call `rewriteParentModule(input.code, relPath, extractions, originalImports)`
   f. For each extraction that is NOT sync$, call `generateSegmentCode(extraction)`

2. Build the `modules` array:
   - First entry: parent module
     ```typescript
     {
       path: relPath,
       isEntry: true,
       code: parentRewriteResult.code,
       map: null,  // Source maps deferred
       segment: null,
       origPath: input.path
     }
     ```
   - Then each segment module:
     ```typescript
     {
       path: extraction.canonicalFilename + extraction.extension,
       isEntry: false,
       code: segmentCode,
       map: null,
       segment: {
         origin: extraction.origin,
         name: extraction.symbolName,
         entry: null,  // Phase 5 handles entry strategies
         displayName: extraction.displayName,
         hash: extraction.hash,
         canonicalFilename: extraction.canonicalFilename,
         extension: extraction.extension,
         parent: extraction.parent,
         ctxKind: extraction.ctxKind,
         ctxName: extraction.ctxName,
         captures: extraction.captures,
         loc: extraction.loc
       },
       origPath: null
     }
     ```

3. Collect diagnostics (empty array for Phase 2; Phase 6 adds diagnostic emission)

4. Return `{ modules, diagnostics: [], isTypeScript, isJsx }`

**Path handling:**
- Use `pathe` for path normalization (forward slashes)
- `relPath` should be the path relative to srcDir, or just the filename if no srcDir relationship
- `canonicalFilename` is the symbolName (used as the segment's file name in import paths)

**Options not yet used (deferred to later phases):**
- `entryStrategy` -> Phase 5
- `mode` -> Phase 5
- `stripExports` -> Phase 5
- `isServer` -> Phase 5
- `transpileTs` / `transpileJsx` -> Handled via oxc-transform when generating segment code (if needed)
- `sourceMaps` -> Deferred (v2)

For Phase 2, the function should work with default settings (smart entry strategy, prod mode). Options that affect behavior in later phases should be accepted but ignored.

Use Phase 1's batch runner to validate transformModule() against real snapshots.

```typescript
import { describe, it, expect } from 'vitest';
import { readFileSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { parseSnapshot } from '../../src/testing/snapshot-parser.js';
import { compareAst } from '../../src/testing/ast-compare.js';
import { transformModule } from '../../src/optimizer/transform.js';
import { getSnapshotFiles } from '../../src/testing/batch-runner.js';

const __dirname = dirname(fileURLToPath(import.meta.url));
const SNAP_DIR = join(__dirname, '../../match-these-snaps');

describe('snapshot batch validation', () => {
  const files = getSnapshotFiles(SNAP_DIR);
  
  // Start with a targeted subset of "simple" snapshots that only test extraction
  // (no captures, no JSX transforms, no signal wrapping, no loops)
  const simpleSnapshots = [
    'qwik_core__test__example_1',
    'qwik_core__test__example_2',
    'qwik_core__test__rename_builder_io',
    // Add more as they pass
  ];
  
  for (const snapName of simpleSnapshots) {
    const snapFile = files.find(f => f.includes(snapName));
    if (!snapFile) continue;
    
    it(`matches ${snapName}`, () => {
      const content = readFileSync(snapFile, 'utf-8');
      const parsed = parseSnapshot(content);
      if (!parsed.input) return; // Skip snapshots without INPUT
      
      const result = transformModule({
        input: [{ path: parsed.frontmatter?.filename || 'test.tsx', code: parsed.input }],
        srcDir: '.',
      });
      
      // Compare parent module
      const parentModule = result.modules[0];
      if (parsed.parentModule) {
        const astResult = compareAst(parsed.parentModule.code, parentModule.code, 'parent');
        expect(astResult.match, `Parent module AST mismatch`).toBe(true);
      }
      
      // Compare segments
      for (const expectedSeg of parsed.segments) {
        const actualSeg = result.modules.find(m => 
          m.segment && m.segment.name === expectedSeg.metadata?.name
        );
        expect(actualSeg, `Missing segment: ${expectedSeg.metadata?.name}`).toBeDefined();
        if (actualSeg) {
          const astResult = compareAst(expectedSeg.code, actualSeg.code, expectedSeg.metadata?.name);
          expect(astResult.match, `Segment ${expectedSeg.metadata?.name} AST mismatch`).toBe(true);
        }
      }
    });
  }
});
```
